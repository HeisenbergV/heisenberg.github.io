```go

func main()
{
    
    new Thread(question)
    
    while {
        startLearn(知识)
    }
}

func question(){
    while(getQuestion())
    {
        询问
        网上搜索
        书籍
    }
}
// error 表示遇到问题
func startLearn(知识)
{
    while(知识)
    {
        if(error)
        {
            addQuestionQueue(error)
            continue
        }

        完成一次学习进度

        归纳总结

        if(需要实践吗)
        {
            while(实践)
            {
                if(error)
                {
                    addQuestionQueue(error)
                    continue
                }
                完成实践
                continue
            }
        }
        else
        {
            脑海模拟
        }

        完善总结
        break
    }
}

```
